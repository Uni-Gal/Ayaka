22/// The plugin module with high-level interfaces.
23pub struct Module<M: RawModule> {
24    module: PluginModule<M>,
25}
26
27impl<M: RawModule> Module<M> {
28    fn new(module: M) -> Self {
29        Self {
30            module: PluginModule::new(module),
31        }
32    }
33
34    /// Gets the [`PluginType`].
35    pub fn plugin_type(&self) -> Result<PluginType> {
36        self.module.call("plugin_type", ())
37    }
38
39    /// Processes [`Action`] in action plugin.
40    pub fn process_action(&self, ctx: ActionProcessContextRef) -> Result<ActionProcessResult> {
41        self.module.call("process_action", (ctx,))
42    }
43
44    /// Calls a custom command in the text plugin.
45    pub fn dispatch_text(
46        &self,
47        name: &str,
48        args: &[String],
49        ctx: TextProcessContextRef,
50    ) -> Result<TextProcessResult> {
51        self.module.call(name, (args, ctx))
52    }
53
54    /// Calls a custom command in the line plugin.
55    pub fn dispatch_line(
56        &self,
57        name: &str,
58        ctx: LineProcessContextRef,
59    ) -> Result<LineProcessResult> {
60        self.module.call(name, (ctx,))
61    }
62
63    /// Processes [`Game`] when opening the config file.
64    pub fn process_game(&self, ctx: GameProcessContextRef) -> Result<GameProcessResult> {
65        self.module.call("process_game", (ctx,))
66    }
67}

69/// The plugin runtime.
70pub struct Runtime<M: RawModule + Send + Sync + 'static> {
71    modules: HashMap<String, Module<M>>,
72    action_modules: Vec<String>,
73    text_modules: HashMap<String, String>,
74    line_modules: HashMap<String, String>,
75    game_modules: Vec<String>,
76}
77
78/// The load status of [`Runtime`].
79#[derive(Debug, Clone)]
80pub enum LoadStatus {
81    /// Start creating the engine.
82    CreateEngine,
83    /// Loading the plugin.
84    LoadPlugin(String, usize, usize),
85}
86
87impl<M: RawModule + Send + Sync + 'static> Runtime<M> {
88    /// Load plugins from specific directory and plugin names.
89    ///
90    /// The actual load folder will be `rel_to.join(dir)`.
91    ///
92    /// If `names` is empty, all WASM files will be loaded.
93    #[stream(LoadStatus, lifetime = 'a)]
94    pub async fn load<'a>(
95        dir: impl AsRef<str> + 'a,
96        root_path: &'a VfsPath,
97        names: &'a [impl AsRef<str>],
98        mut store: M::Linker,
99    ) -> Result<Arc<Self>> {
100        let path = root_path.join(dir)?;
101        let paths = Self::find_plugins(&path, names)?;
102
103        yield LoadStatus::CreateEngine;
104        let handle = Arc::new(RwLock::new(Weak::new()));
105        log_interop::register(&mut store)?;
106        plugin_interop::register(&mut store, handle.clone())?;
107        fs_interop::register(&mut store, root_path)?;
108        rand_interop::register(&mut store)?;
109        script_interop::register(&mut store)?;
110        let mut runtime = Self::new();
111
112        let total_len = paths.len();
113        for (i, (name, p)) in paths.into_iter().enumerate() {
114            yield LoadStatus::LoadPlugin(name.clone(), i, total_len);
115            let mut buf = vec![];
116            p.open_file()?.read_to_end(&mut buf)?;
117            let module = Module::new(store.create(&buf)?);
118            runtime.insert_module(name, module)?;
119        }
120        let runtime = Arc::new(runtime);
121        *handle.write().unwrap() = Arc::downgrade(&runtime);
122        Ok(runtime)
123    }
124
125    fn new() -> Self {
126        Self {
127            modules: HashMap::default(),
128            action_modules: vec![],
129            text_modules: HashMap::default(),
130            line_modules: HashMap::default(),
131            game_modules: vec![],
132        }
133    }
134
135    fn find_plugins(path: &VfsPath, names: &[impl AsRef<str>]) -> Result<Vec<(String, VfsPath)>> {
136        let paths = if names.is_empty() {
137            path.read_dir()?
138                .filter_map(|p| {
139                    if p.is_file().unwrap_or_default()
140                        && p.extension().unwrap_or_default() == "wasm"
141                    {
142                        let name = p
143                            .filename()
144                            .strip_suffix(".wasm")
145                            .unwrap_or_default()
146                            .to_string();
147                        Some((name, p))
148                    } else {
149                        None
150                    }
151                })
152                .collect::<Vec<_>>()
153        } else {
154            names
155                .iter()
156                .filter_map(|name| {
157                    let name = name.as_ref();
158                    let p = path.join(format!("{name}.wasm")).ok()?;
159                    if p.exists().unwrap_or_default() {
160                        Some((name.to_string(), p))
161                    } else {
162                        None
163                    }
164                })
165                .collect::<Vec<_>>()
166        };
167        Ok(paths)
168    }
169
170    fn insert_module(&mut self, name: String, module: Module<M>) -> Result<()> {
171        let plugin_type =
172            unwrap_or_default_log!(module.plugin_type(), "Cannot determine module type");
173        if plugin_type.action {
174            self.action_modules.push(name.clone());
175        }
176        for cmd in plugin_type.text {
177            let res = self.text_modules.insert(cmd.clone(), name.clone());
178            if let Some(old_module) = res {
179                log::warn!("Text command `{cmd}` is overrided by \"{name}\" over \"{old_module}\"");
180            }
181        }
182        for cmd in plugin_type.line {
183            let res = self.line_modules.insert(cmd.clone(), name.clone());
184            if let Some(old_module) = res {
185                log::warn!("Line command `{cmd}` is overrided by \"{name}\" over \"{old_module}\"");
186            }
187        }
188        if plugin_type.game {
189            self.game_modules.push(name.clone());
190        }
191        self.modules.insert(name, module);
192        Ok(())
193    }
194
195    /// Gets module from name.
196    pub fn module(&self, key: &str) -> Option<&Module<M>> {
197        self.modules.get(key)
198    }
199
200    /// Iterates action modules.
201    pub fn action_modules(&self) -> impl Iterator<Item = &Module<M>> {
202        self.action_modules
203            .iter()
204            .filter_map(|key| self.module(key))
205    }
206
207    /// Gets text module from command.
208    pub fn text_module(&self, cmd: &str) -> Option<&Module<M>> {
209        self.text_modules.get(cmd).and_then(|key| self.module(key))
210    }
211
212    /// Gets line module from command.
213    pub fn line_module(&self, cmd: &str) -> Option<&Module<M>> {
214        self.line_modules.get(cmd).and_then(|key| self.module(key))
215    }
216
217    /// Iterates game modules.
218    pub fn game_modules(&self) -> impl Iterator<Item = &Module<M>> {
219        self.game_modules.iter().filter_map(|key| self.module(key))
220    }
221}

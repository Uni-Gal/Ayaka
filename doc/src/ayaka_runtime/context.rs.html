27/// The open status when creating [`Context`].
28#[derive(Debug, Clone, PartialEq, Eq, Serialize)]
29#[serde(tag = "t", content = "data")]
30pub enum OpenStatus {
31    /// Start loading config file.
32    LoadProfile,
33    /// Start creating plugin runtime.
34    CreateRuntime,
35    /// Loading the plugin.
36    LoadPlugin(String, usize, usize),
37    /// Executing game plugins.
38    GamePlugin,
39    /// Loading the resources.
40    LoadResource,
41    /// Loading the paragraphs.
42    LoadParagraph,
43}
44
45impl From<LoadStatus> for OpenStatus {
46    fn from(value: LoadStatus) -> Self {
47        match value {
48            LoadStatus::CreateEngine => Self::CreateRuntime,
49            LoadStatus::LoadPlugin(name, i, len) => Self::LoadPlugin(name, i, len),
50        }
51    }
52}

66    fn open_fs_from_paths(paths: &'_ [impl AsRef<Path>]) -> Result<(VfsPath, Cow<'_, str>)> {
67        let (root_path, filename) = if paths.len() == 1 {
68            let path = paths[0].as_ref();
69            let ext = path.extension().unwrap_or_default();
70            if ext == "yaml" {
71                let root_path = path
72                    .parent()
73                    .ok_or_else(|| anyhow!("Cannot get parent from input path."))?;
74                (
75                    VfsPath::from(PhysicalFS::new(root_path)),
76                    path.file_name().unwrap_or_default().to_string_lossy(),
77                )
78            } else if ext == "ayapack" {
79                (TarFS::new_mmap(path)?.into(), "config.yaml".into())
80            } else {
81                bail!("Cannot determine filesystem.")
82            }
83        } else {
84            let files = paths
85                .iter()
86                .rev()
87                .map(|path| TarFS::new_mmap(path.as_ref()).map(VfsPath::from))
88                .collect::<Result<Vec<_>, _>>()?;
89            (OverlayFS::new(&files).into(), "config.yaml".into())
90        };
91        Ok((root_path, filename))
92    }

94    /// Open a context with config paths.
95    ///
96    /// If the input `paths` contains only one element, it may be a YAML or an FRFS file.
97    /// If the input `paths` contains many element, they should all be FRFS files,
98    /// and the latter one will override the former one.
99    pub fn with_paths(
100        self,
101        paths: &'_ [impl AsRef<Path>],
102    ) -> Result<ContextBuilderWithPaths<'_, M>> {
103        if paths.is_empty() {
104            bail!("At least one path should be input.");
105        }
106        let (root_path, filename) = Self::open_fs_from_paths(paths)?;
107        Ok(ContextBuilderWithPaths {
108            root_path,
109            filename,
110            frontend: self.frontend,
111            linker: self.linker,
112        })
113    }

144impl<M: RawModule + Send + Sync + 'static> Context<M> {
145    #[stream(OpenStatus, lifetime = 'a)]
146    async fn open<'a>(
147        root_path: VfsPath,
148        filename: impl AsRef<str> + 'a,
149        frontend: FrontendType,
150        linker: M::Linker,
151    ) -> Result<Self> {
152        yield OpenStatus::LoadProfile;
153        let file = root_path.join(filename.as_ref())?.open_file()?;
154        let mut config: GameConfig = serde_yaml::from_reader(file)?;
155        let runtime = {
156            let runtime = Runtime::load(
157                &config.plugins.dir,
158                &root_path,
159                &config.plugins.modules,
160                linker,
161            );
162            let mut runtime = pin!(runtime);
163            while let Some(load_status) = runtime.next().await {
164                yield load_status.into();
165            }
166            runtime.await?
167        };
168
169        yield OpenStatus::GamePlugin;
170        Self::preprocess_game(&mut config, &runtime)?;
171
172        yield OpenStatus::LoadResource;
173        let res = Self::load_resource(&config, &root_path)?;
174
175        yield OpenStatus::LoadParagraph;
176        let paras = Self::load_paragraph(&config, &root_path)?;
177
178        Ok(Self {
179            game: Game { config, paras, res },
180            root_path,
181            frontend,
182            runtime,
183            ctx: RawContext::default(),
184            switches: vec![],
185            vars: VarMap::default(),
186        })
187    }

189    fn preprocess_game(config: &mut GameConfig, runtime: &Runtime<M>) -> Result<()> {
190        for module in runtime.game_modules() {
191            let ctx = GameProcessContextRef {
192                title: &config.title,
193                author: &config.author,
194                props: &config.props,
195            };
196            let res = module.process_game(ctx)?;
197            for (key, value) in res.props {
198                config.props.insert(key, value);
199            }
200        }
201        Ok(())
202    }

204    fn load_resource(
205        config: &GameConfig,
206        root_path: &VfsPath,
207    ) -> Result<HashMap<Locale, HashMap<String, RawValue>>> {
208        let mut res = HashMap::new();
209        if let Some(res_path) = &config.res {
210            let res_path = root_path.join(res_path)?;
211            for p in res_path.read_dir()? {
212                if p.is_file()? && p.extension().unwrap_or_default() == "yaml" {
213                    if let Ok(loc) = p
214                        .filename()
215                        .strip_suffix(".yaml")
216                        .unwrap_or_default()
217                        .parse::<Locale>()
218                    {
219                        let r = p.open_file()?;
220                        let r = serde_yaml::from_reader(r)?;
221                        res.insert(loc, r);
222                    }
223                }
224            }
225        }
226        Ok(res)
227    }

229    fn load_paragraph(
230        config: &GameConfig,
231        root_path: &VfsPath,
232    ) -> Result<HashMap<Locale, HashMap<String, Vec<Paragraph>>>> {
233        let mut paras = HashMap::new();
234        let paras_path = root_path.join(&config.paras)?;
235        for p in paras_path.read_dir()? {
236            if p.is_dir()? {
237                if let Ok(loc) = p.filename().parse::<Locale>() {
238                    let mut paras_map = HashMap::new();
239                    for p in p.read_dir()? {
240                        if p.is_file()? && p.extension().unwrap_or_default() == "yaml" {
241                            let key = p
242                                .filename()
243                                .strip_suffix(".yaml")
244                                .unwrap_or_default()
245                                .to_string();
246                            let para = p.open_file()?;
247                            let para = serde_yaml::from_reader(para)?;
248                            paras_map.insert(key, para);
249                        }
250                    }
251                    paras.insert(loc, paras_map);
252                }
253            }
254        }
255        Ok(paras)
256    }

311        assert!((0..self.switches.len()).contains(&i));
312        assert!(self.switches[i]);
313        self.ctx
314            .locals
315            .insert("?".to_string(), RawValue::Num(i as i64));
316        for i in 0..self.switches.len() {

321    fn parse_text(&self, loc: &Locale, text: &Text, ctx: &RawContext) -> Result<ActionText> {
322        let mut action = ActionText::default();
323        action.ch_key = text.ch_tag.clone();
324        action.character = text.ch_alias.clone().or_else(|| {
325            self.find_res(
326                loc,
327                &format!("ch_{}", action.ch_key.as_deref().unwrap_or_default()),
328            )
329            .map(|value| value.get_str().into_owned())
330        });
331        for sub_text in &text.sub_texts {
332            let mut sub_action = self.parse_sub_text(sub_text, Some(loc), &ctx.locals)?;
333            action.text.append(&mut sub_action.text);
334        }
335        Ok(action)
336    }

338    fn parse_sub_text(
339        &self,
340        sub_text: &SubText,
341        loc: Option<&Locale>,
342        locals: &VarMap,
343    ) -> Result<ActionText> {
344        let mut action = ActionText::default();
345        match sub_text {
346            SubText::Char(c) => action.push_back_chars(c.to_string()),
347            SubText::Str(s) => action.push_back_chars(s),
348            SubText::Cmd(cmd, args) => {
349                let mut arg_strings = vec![];
350                for arg in args {
351                    let sub_action = self.parse_sub_text(arg, loc, locals)?;
352                    arg_strings.push(sub_action.to_string());
353                }
354                match cmd.as_str() {
355                    "res" => {
356                        if let Some(loc) = loc {
357                            if arg_strings.len() != 1 {
358                                log::warn!("Invalid parameter count for `res`: {}", args.len())
359                            }
360                            if let Some(n) = arg_strings.first() {
361                                if let Some(value) = self.find_res(loc, n) {
362                                    action.push_back_block(value.get_str())
363                                } else {
364                                    log::warn!("Cannot find resource {n}");
365                                }
366                            }
367                        }
368                    }
369                    "var" => {
370                        if arg_strings.len() != 1 {
371                            log::warn!("Invalid parameter count for `var`: {}", args.len())
372                        }
373                        if let Some(n) = arg_strings.first() {
374                            if let Some(value) = locals.get(n) {
375                                action.push_back_block(value.get_str())
376                            } else {
377                                log::warn!("Cannot find variable {n}")
378                            }
379                        }
380                    }
381                    _ => {
382                        if let Some(module) = self.runtime.text_module(cmd) {
383                            let ctx = TextProcessContextRef {
384                                game_props: &self.game.config.props,
385                                frontend: self.frontend,
386                            };
387                            let mut res = module.dispatch_text(cmd, &arg_strings, ctx)?;
388                            action.text.append(&mut res.text.text);
389                            action.vars.extend(res.text.vars);
390                        }
391                    }
392                }
393            }
394        }
395        Ok(action)
396    }

408    fn process_line(&mut self, t: Line) -> Result<()> {
409        match t {
410            Line::Empty | Line::Text(_) => {}
411            Line::Switch { switches } => {
412                self.switches.clear();
413                for i in 0..switches.len() {
414                    let enabled = self
415                        .ctx
416                        .locals
417                        .get(&i.to_string())
418                        .unwrap_or(&RawValue::Unit);
419                    let enabled = if let RawValue::Unit = enabled {
420                        true
421                    } else {
422                        enabled.get_bool()
423                    };
424                    self.switches.push(enabled);
425                }
426            }
427            Line::Custom(props) => {
428                self.vars.clear();
429                let cmd = props.iter().next().map(|(key, _)| key);
430                if let Some(cmd) = cmd {
431                    if let Some(module) = self.runtime.line_module(cmd) {
432                        let ctx = LineProcessContextRef {
433                            game_props: &self.game.config.props,
434                            frontend: self.frontend,
435                            ctx: &self.ctx,
436                            props: &props,
437                        };
438                        let res = module.dispatch_line(cmd, ctx)?;
439                        self.ctx.locals.extend(res.locals);
440                        self.vars.extend(res.vars);
441                    } else {
442                        bail!("Cannot find command {}", cmd)
443                    }
444                }
445            }
446        }
447        Ok(())
448    }

450    fn merge_action(&self, action: Fallback<Action>) -> Result<Action> {
451        match action.unzip() {
452            (None, None) => Ok(Action::default()),
453            (Some(action), None) | (None, Some(action)) => Ok(action),
454            (Some(action), Some(action_base)) => match (action, action_base) {
455                (Action::Text(action), Action::Text(action_base)) => {
456                    let action = Fallback::new(Some(action), Some(action_base));
457                    let action = action.spec();
458                    Ok(Action::Text(ActionText {
459                        text: action.text.and_any().unwrap_or_default(),
460                        ch_key: action.ch_key.flatten().fallback(),
461                        character: action.character.flatten().fallback(),
462                        vars: action.vars.and_any().unwrap_or_default(),
463                    }))
464                }
465                (Action::Switches(mut switches), Action::Switches(switches_base)) => {
466                    for (item, item_base) in switches.iter_mut().zip(switches_base) {
467                        item.enabled = item_base.enabled;
468                    }
469                    Ok(Action::Switches(switches))
470                }
471                (Action::Custom(mut vars), Action::Custom(vars_base)) => {
472                    vars.extend(vars_base);
473                    Ok(Action::Custom(vars))
474                }
475                _ => bail!("Mismatching action type"),
476            },
477        }
478    }

480    fn process_action_text(&self, ctx: &RawContext, action: &mut ActionText) -> Result<()> {
481        for module in self.runtime.action_modules() {
482            let ctx = ActionProcessContextRef {
483                game_props: &self.game.config.props,
484                frontend: self.frontend,
485                ctx,
486                action,
487            };
488            *action = module.process_action(ctx)?.action;
489        }
490        while let Some(act) = action.text.back() {
491            if act.as_str().trim().is_empty() {
492                action.text.pop_back();
493            } else {
494                break;
495            }
496        }
497        while let Some(act) = action.text.front() {
498            if act.as_str().trim().is_empty() {
499                action.text.pop_front();
500            } else {
501                break;
502            }
503        }
504        Ok(())
505    }

507    /// Get the [`Action`] from [`Locale`] and [`RawContext`].
508    pub fn get_action(&self, loc: &Locale, ctx: &RawContext) -> Result<Action> {
509        let cur_text = self
510            .game
511            .find_para_fallback(loc, &ctx.cur_base_para, &ctx.cur_para)
512            .map(|p| p.texts.get(ctx.cur_act))
513            .flatten();
514
515        let action = cur_text
516            .map(|t| match t {
517                Line::Text(t) => self.parse_text(loc, t, ctx).map(Action::Text).ok(),
518                Line::Switch { switches } => Some(Action::Switches(self.parse_switches(switches))),
519                // The real vars will be filled in `merge_action`.
520                Line::Custom(_) => Some(Action::Custom(self.vars.clone())),
521                _ => None,
522            })
523            .flatten();
524
525        let mut act = self.merge_action(action)?;
526        if let Action::Text(act) = &mut act {
527            self.process_action_text(ctx, act)?;
528        }
529        Ok(act)
530    }

532    /// Step to next line.
533    pub fn next_run(&mut self) -> Option<RawContext> {
534        let cur_text_base = loop {
535            let cur_para = self.current_paragraph(&self.game.config.base_lang);
536            let cur_text = self.current_text(&self.game.config.base_lang);
537            match (cur_para.is_some(), cur_text.is_some()) {
538                (true, true) => break cur_text,
539                (true, false) => {
540                    self.ctx.cur_para = cur_para
541                        .and_then(|p| p.next.as_ref())
542                        .map(|text| unwrap_or_default_log!(self.call(text), "Cannot get next para"))
543                        .unwrap_or_default();
544                    self.ctx.cur_act = 0;
545                }
546                (false, _) => {
547                    if self.ctx.cur_base_para == self.ctx.cur_para {
548                        if !self.ctx.cur_para.is_empty() {
549                            error!(
550                                "Cannot find paragraph \"{}\"",
551                                self.ctx.cur_para.escape_default()
552                            );
553                        }
554                        return None;
555                    } else {
556                        self.ctx.cur_base_para = self.ctx.cur_para.clone();
557                    }
558                }
559            }
560        };
561
562        let ctx = cur_text_base.cloned().map(|t| {
563            unwrap_or_default_log!(self.process_line(t), "Parse line error");
564            self.ctx.clone()
565        });
566        self.ctx.cur_act += 1;
567        ctx
568    }

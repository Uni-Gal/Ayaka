1use super::Runtime;
2use anyhow::{bail, Result};
3use ayaka_plugin::{Linker, LinkerHandle, RawModule};
4use std::{
5    collections::HashMap,
6    sync::{Arc, RwLock, Weak},
7};
8
9pub fn register<M: RawModule + Send + Sync + 'static>(
10    store: &mut impl Linker<M>,
11    handle: Arc<RwLock<Weak<Runtime<M>>>>,
12) -> Result<()> {
13    let h = handle.clone();
14    let modules_func = store.wrap(move |_: ()| {
15        if let Some(this) = h.read().unwrap().upgrade() {
16            Ok(this.modules.keys().cloned().collect::<Vec<_>>())
17        } else {
18            bail!("Runtime hasn't been initialized.")
19        }
20    });
21    let h = handle;
22    let call_func = store.wrap_with(
23        move |mut handle, (module, name, args): (String, String, Vec<u8>)| {
24            if let Some(this) = h.read().unwrap().upgrade() {
25                Ok(handle.call(
26                    this.modules[&module].module.inner(),
27                    &name,
28                    &args,
29                    |slice| Ok(slice.to_vec()),
30                )?)
31            } else {
32                bail!("Runtime hasn't been initialized.")
33            }
34        },
35    );
36    store.import(
37        "plugin",
38        HashMap::from([
39            ("__modules".to_string(), modules_func),
40            ("__call".to_string(), call_func),
41        ]),
42    )?;
43    Ok(())
44}
